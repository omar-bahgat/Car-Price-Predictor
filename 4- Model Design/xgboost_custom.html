def hybrid_obj(y_true, y_pred, threshold=0.08):
    err = y_pred - y_true
    grad = np.where(np.abs(err) < threshold, 3 * err**2, 2 * err)
    hess = np.where(np.abs(err) < threshold, 6 * err, 2)
    return grad, hess
